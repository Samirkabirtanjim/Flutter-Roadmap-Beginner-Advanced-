# Flutter Random Number app Example  
This Flutter app demonstrates how to use **get random number**.

---

```dart
import 'package:flutter/material.dart';
import 'package:check/logics/randomnumber.dart';

void main() => runApp(const RandomNumber());
```
```dart
import 'dart:math';
import 'package:flutter/material.dart';

class RandomNumber extends StatelessWidget {
  const RandomNumber({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      home: Scaffold(
        appBar: AppBar(
          title: const Text(
            'Random Number Generator',
            style: TextStyle(color: Colors.white),
          ),
          backgroundColor: Colors.teal,
        ),
        body: Center(
          child: Text(
            "Random Number is: ${getNumbers()}",
            style: TextStyle(fontSize: 24),
          ),
        ),
      ),
    );
  }
}

int getNumbers() {
  Random random = Random();
  return random.nextInt(100); // Generates 0â€“99
}
```
