# RowandColumn Flutter Demo

This Flutter app showcases **Bangladesh images, flag description, and famous landmarks**. It demonstrates the use of:

- `Column`, `Row`, and `Expanded` for layouts
- `ClipRRect` for rounded images
- `Padding` for spacing
- `Text` styling and alignment
- `BoxFit.cover` for image scaling

## Flutter Code

```dart
import 'package:flutter/material.dart';

class RowCheck extends StatelessWidget {
  const RowCheck({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      home: Scaffold(
        backgroundColor: Color(0XFFF9FBE7),
        appBar: AppBar(
          title: Text("Bangladesh", style: TextStyle(color: Colors.black)),
          backgroundColor: Color(0XFFC5E1A5),
        ),
        body: Column(
          children: [
            // Full-width banner image
            Padding(
              padding: const EdgeInsets.all(1.0),
              child: Image(
                image: AssetImage("images/banthree.png"),
                width: double.infinity,
                height: 250,
                fit: BoxFit.cover,
              ),
            ),

            // Flag description
            Padding(
              padding: const EdgeInsets.symmetric(horizontal: 8, vertical: 8),
              child: Text(
                "Bangladesh's flag features a bottle-green background representing the nation's lush greenery and vitality. "
                "A red disc, shifted slightly toward the hoist, symbolizes the rising sun of independence and the blood "
                "sacrificed by martyrs during the 1971 Liberation War. Designed by Quamrul Hassan, it reflects the country's "
                "beauty and revolutionary spirit.",
                textAlign: TextAlign.justify,
                style: TextStyle(
                  fontSize: 15,
                  height: 1.2,
                  color: Colors.black87,
                ),
              ),
            ),

            // Row of images with captions
            Padding(
              padding: const EdgeInsets.all(1),
              child: Row(
                children: [
                  // First image + caption
                  Expanded(
                    child: Column(
                      children: [
                        ClipRRect(
                          borderRadius: BorderRadius.circular(6),
                          child: Image.asset(
                            "images/banone.png",
                            fit: BoxFit.cover,
                          ),
                        ),
                        SizedBox(height: 2),
                        Padding(
                          padding: const EdgeInsets.symmetric(horizontal: 8, vertical: 8),
                          child: Text(
                            "A glimpse into the past at the UNESCO World Heritage Sixty Dome Mosque.",
                            textAlign: TextAlign.justify,
                            style: TextStyle(
                              fontSize: 14,
                              height: 1.2,
                              color: Colors.black87,
                            ),
                          ),
                        ),
                      ],
                    ),
                  ),

                  SizedBox(width: 1.0),

                  // Second image + caption
                  Expanded(
                    child: Column(
                      children: [
                        ClipRRect(
                          borderRadius: BorderRadius.circular(6),
                          child: Image.asset(
                            "images/tiger.png",
                            fit: BoxFit.cover,
                          ),
                        ),
                        SizedBox(height: 2),
                        Padding(
                          padding: const EdgeInsets.symmetric(horizontal: 8, vertical: 8),
                          child: Text(
                            "The King of the Mangroves: A Royal Bengal Tiger in the Sundarbans.",
                            textAlign: TextAlign.justify,
                            style: TextStyle(
                              fontSize: 14,
                              height: 1.2,
                              color: Colors.black87,
                            ),
                          ),
                        )
                      ],
                    ),
                  ),
                ],
              ),
            ),
          ],
        ),
      ),
    );
  }
}
