## Add Material Icon in Flutter

This Flutter project demonstrates how to use the IconButton widget with custom size, colors, splash effects, and an AppBar inside a simple MaterialApp.

import 'package:flutter/material.dart';

class Iconflutter extends StatelessWidget {
  const Iconflutter({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      home: Scaffold(
        backgroundColor: Colors.yellowAccent,
        appBar: AppBar(
          title: Text("Icon", style: TextStyle(color: Colors.white)),
          backgroundColor: Colors.teal,
        ),
        body: Center(
          child: IconButton(
            icon: Icon(Icons.one_x_mobiledata),
            onPressed: () {},
            iconSize: 100,
            color: Colors.black,
            splashColor: Colors.teal,
            highlightColor: Colors.tealAccent,
          ),
        ),
      ),
    );
  }
}

## Font Awesome Icons in Flutter

```dart
import 'package:flutter/material.dart';
import 'package:font_awesome_flutter/font_awesome_flutter.dart';

class Iconflutter extends StatelessWidget {
  const Iconflutter({super.key});

  @override
  Widget build(BuildContext context) {
    return MaterialApp(
      debugShowCheckedModeBanner: false,
      home: Scaffold(
        backgroundColor: Colors.lightBlueAccent,
        appBar: AppBar(
          title: Text("Icon", style: TextStyle(color: Colors.white)),
          backgroundColor: Colors.teal,
        ),
        body: Center(
          child: IconButton(
            icon: Icon(FontAwesomeIcons.addressBook),
            onPressed: () {},
            iconSize: 100,
            color: Colors.white,
            splashColor: Colors.teal,
            highlightColor: Colors.tealAccent,
          ),
        ),
      ),
    );
  }
}
```
